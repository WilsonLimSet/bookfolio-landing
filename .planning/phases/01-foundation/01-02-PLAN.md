import Foundation

struct Profile: Codable, Identifiable {
    let id: UUID
    let username: String
    let bio: String?
    let avatarUrl: String?
    let instagram: String?
    let twitter: String?
    let referralCode: String?
    let readingGoal2025: Int?
    let updatedAt: Date?

    enum CodingKeys: String, CodingKey {
        case id, username, bio, instagram, twitter
        case avatarUrl = "avatar_url"
        case referralCode = "referral_code"
        case readingGoal2025 = "reading_goal_2025"
        case updatedAt = "updated_at"
    }
}
```

struct OLSearchResponse: Codable {

struct OLSearchResult: Codable, Identifiable {
    var id: String { key }
    let key: String              // "/works/OL12345W"
    let title: String
    let authorName: [String]?    // "author_name" in JSON
    let coverId: Int?            // "cover_i" in JSON
    let firstPublishYear: Int?   // "first_publish_year"
    let numberOfPagesMedian: Int? // "number_of_pages_median"
    let subject: [String]?

    enum CodingKeys: String, CodingKey {
        case key, title, subject
        case authorName = "author_name"
        case coverId = "cover_i"
        case firstPublishYear = "first_publish_year"
        case numberOfPagesMedian = "number_of_pages_median"
    }
}
```

struct OLWork: Codable {

enum OLCoverSize: String {

        URL(string: "https://covers.openlibrary.org/b/id/\(self)-\(size.rawValue).jpg")

}
