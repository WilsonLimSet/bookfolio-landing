```swift
struct ProfileStats {
    let fictionCount: Int

    let wantToReadCount: Int

}
```

Static methods (no @Published state needed — ProfileView will call these directly):

1. `static func fetchProfile(username: String) async throws -> Profile`:
   - `.from("profiles").select().eq("username", value: username).single().execute()` decoded as Profile

2. `static func fetchProfile(userId: UUID) async throws -> Profile`:
   - `.from("profiles").select().eq("id", value: userId.uuidString).single().execute()` decoded as Profile

3. `static func fetchStats(userId: UUID) async throws -> ProfileStats`:
   Run all these queries concurrently with async let:

   - Fiction count: `from("user_books").select("id", head: true, count: .exact).eq("user_id", ...).eq("category", value: "fiction")`
   - Nonfiction count: same but category = "nonfiction"
   - Want-to-read count: `from("want_to_read").select("user_id", head: true, count: .exact).eq("user_id", ...)`
   - Currently reading count: similar on currently_reading
   - Books this year: `from("user_books").select("id", head: true, count: .exact).eq("user_id", ...).gte("finished_at", value: startOfYear).lte("finished_at", value: endOfYear)`
     - startOfYear: ISO8601 string for Jan 1 of current year
     - endOfYear: ISO8601 string for Dec 31 of current year
   - Followers: `from("follows").select("follower_id", head: true, count: .exact).eq("following_id", ...)`
   - Following: `from("follows").select("following_id", head: true, count: .exact).eq("follower_id", ...)`
   - Favorites: `from("favorite_books").select().eq("user_id", ...).order("position").limit(4)` decoded as [FavoriteBook]
   - Referrals: `from("referrals").select("id", head: true, count: .exact).eq("referrer_id", ...)`
   - Rank: `try? await supabase.rpc("get_user_book_rank", params: ["target_user_id": userId.uuidString]).execute().value as Int?` — wrap in try? so missing rank returns nil

   For week streak calculation:
   - Fetch activities from last 12 weeks: `from("activity").select("created_at").eq("user_id", ...).gte("created_at", value: twelveWeeksAgo)`
   - Group by ISO week in Swift, count consecutive weeks from current week backwards

   IMPORTANT: supabase-swift count queries use `.select("id", head: true, count: .exact)` and return count in response. Access via `response.count`.
   IMPORTANT: For async let, you need each query in a separate async let binding, then await all at the end.
   IMPORTANT: Date formatting for Supabase queries should use ISO8601DateFormatter.

4. `static func fetchFollowers(userId: UUID) async throws -> [(Profile, Date)]`:
   - Query follows WHERE following_id = userId
   - Get follower profile IDs
   - Query profiles for those IDs
   - Return array of (Profile, followDate)

5. `static func fetchFollowing(userId: UUID) async throws -> [(Profile, Date)]`:
   - Query follows WHERE follower_id = userId
   - Get following profile IDs
   - Query profiles for those IDs

6. `static func isFollowing(currentUserId: UUID, targetUserId: UUID) async throws -> Bool`:
   - Query follows WHERE follower_id = currentUserId AND following_id = targetUserId
   - Return count > 0

7. `static func follow(currentUserId: UUID, targetUserId: UUID) async throws`:
   - Insert into follows(follower_id, following_id)
   - Insert into activity(user_id, action_type: "followed", target_user_id)
   - Insert into notifications(user_id: targetUserId, type: "follow", from_user_id: currentUserId)

8. `static func unfollow(currentUserId: UUID, targetUserId: UUID) async throws`:
   - Delete from follows WHERE follower_id AND following_id
  </action>
