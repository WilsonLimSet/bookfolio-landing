Native Google Sign-In approach (recommended by Supabase for mobile):
1. User taps "Continue with Google" → native OS dialog (no web browser)
2. GoogleSignIn SDK returns ID token
3. Pass ID token to supabase.auth.signInWithIdToken(credentials:)
4. Supabase verifies token server-side, creates/returns session
5. authStateChanges fires .signedIn event → AuthService updates state

Required setup:
- SPM: google-sign-in-ios (https://github.com/google/GoogleSignIn-iOS)
- Google Cloud Console: iOS OAuth client ID (bundle ID based)
- Supabase Dashboard: Google provider enabled with iOS client ID in authorized list
- Info.plist: GIDClientID + URL scheme for Google SDK
- Nonce: SHA-256 hashed nonce sent to Google, raw nonce sent to Supabase

supabase-swift API:
```swift
try await supabase.auth.signInWithIdToken(
    credentials: OpenIDConnectCredentials(
        provider: .google,
        idToken: idToken,
        accessToken: accessToken,
        nonce: rawNonce  // unhashed
    )
)
```

Google Sign-In iOS SDK:

```swift
let result = try await GIDSignIn.sharedInstance.signIn(
    withPresenting: viewController,
    hint: nil,
    additionalScopes: nil,
    nonce: hashedNonce  // SHA-256 hashed
)
```

```swift
import Foundation
import GoogleSignIn
import CryptoKit
import Auth  // from supabase-swift

@MainActor
class GoogleAuthService {

    // Generate cryptographically secure nonce
    private func generateNonce() -> String {
        // Use random bytes for nonce
        var randomBytes = [UInt8](repeating: 0, count: 32)
        _ = SecRandomCopyBytes(kSecRandomDefault, randomBytes.count, &randomBytes)
        return randomBytes.map { String(format: "%02x", $0) }.joined()
    }

    // SHA-256 hash of nonce
    private func sha256(_ input: String) -> String {
        let data = Data(input.utf8)
        let hash = SHA256.hash(data: data)
        return hash.compactMap { String(format: "%02x", $0) }.joined()
    }

    func signIn() async throws {
        let rawNonce = generateNonce()
        let hashedNonce = sha256(rawNonce)

        // Get root view controller for presenting Google sign-in
        guard let windowScene = UIApplication.shared.connectedScenes.first as? UIWindowScene,
              let rootViewController = windowScene.windows.first?.rootViewController else {
            throw GoogleAuthError.noRootViewController
        }

        // Present native Google Sign-In dialog
        let result = try await GIDSignIn.sharedInstance.signIn(
            withPresenting: rootViewController,
            hint: nil,
            additionalScopes: nil,
            nonce: hashedNonce
        )

        guard let idToken = result.user.idToken?.tokenString else {
            throw GoogleAuthError.noIdToken
        }

        // Pass to Supabase with raw (unhashed) nonce
        try await supabase.auth.signInWithIdToken(
            credentials: OpenIDConnectCredentials(
                provider: .google,
                idToken: idToken,
                accessToken: result.user.accessToken.tokenString,
                nonce: rawNonce
            )
        )
        // AuthService.authStateChanges will pick up the .signedIn event
    }
}

enum GoogleAuthError: LocalizedError {
    case noRootViewController
    case noIdToken

    var errorDescription: String? {
        switch self {
        case .noRootViewController: return "Could not find root view controller"
        case .noIdToken: return "Google Sign-In did not return an ID token"
        }
    }
}
```

IMPORTANT: The nonce flow is critical — Google gets the HASHED nonce, Supabase gets the RAW nonce. Supabase hashes internally to verify.
IMPORTANT: `signIn(withPresenting:hint:additionalScopes:nonce:)` — the nonce parameter is available in GoogleSignIn v8+.
