supabase-swift auth patterns:
- `supabase.auth.signUp(email:password:data:)` — returns AuthResponse (session nil if email confirmation enabled)
- `supabase.auth.signIn(email:password:)` — returns Session
- `supabase.auth.signOut()` — clears Keychain session
- `supabase.auth.session` — async, gets current session (refreshes if needed)
- `supabase.auth.currentUser` — sync, returns User? from cached session
- `supabase.auth.authStateChanges` — AsyncSequence of (AuthChangeEvent, Session?)
- `.initialSession` event fires on launch with restored session or nil
- Error type: `AuthError.api(message:errorCode:underlyingData:underlyingResponse:)`
- Error codes: "invalid_credentials", "user_already_exists", "email_not_confirmed", "weak_password", "over_request_rate_limit"

Existing web app auth flow:
- Sign-up captures username + email + password
- Username validated against profiles table BEFORE signUp call
- signUp passes username in user_metadata: `data: { username: string }`
- Email confirmation required (magic link with token_hash)
- Sign-in: email + password only
- On success: redirect to /feed
- Profile rows created automatically via Supabase database trigger on auth.users INSERT

1. Create Services/AuthService.swift — ObservableObject managing auth state:

```swift
@MainActor
class AuthService: ObservableObject {
    enum AuthState {
        case loading        // Checking initial session
        case unauthenticated
        case authenticated(User)
        case needsUsername(User)  // OAuth user without username
    }

    @Published var state: AuthState = .loading
    @Published var isLoading = false

    init() {
        Task { await listenForAuthChanges() }
    }

    // Listen to auth state changes via AsyncSequence
    func listenForAuthChanges() async {
        for await (event, session) in supabase.auth.authStateChanges {
            if event == .initialSession || event == .signedIn {
                if let session {
                    await checkUsernameAndSetState(user: session.user)
                } else {
                    state = .unauthenticated
                }
            } else if event == .signedOut {
                state = .unauthenticated
            }
        }
    }

    // Check if user has username in profiles table
    func checkUsernameAndSetState(user: User) async {
        do {
            let profile: Profile = try await supabase.from("profiles")
                .select("id, username")
                .eq("id", value: user.id)
                .single()
                .execute()
                .value
            if profile.username.isEmpty || profile.username == nil {
                state = .needsUsername(user)
            } else {
                state = .authenticated(user)
            }
        } catch {
            // Profile may not exist yet (trigger delay), treat as needs setup
            state = .needsUsername(user)
        }
    }

    // Email/password sign in
    func signIn(email: String, password: String) async throws {
        isLoading = true
        defer { isLoading = false }
        _ = try await supabase.auth.signIn(email: email, password: password)
        // authStateChanges will update state
    }

    // Email/password sign up (with username)
    func signUp(email: String, password: String, username: String) async throws -> Bool {
        isLoading = true
        defer { isLoading = false }

        // Check username availability first
        let existing: [Profile] = try await supabase.from("profiles")
            .select("username")
            .eq("username", value: username)
            .execute()
            .value
        if !existing.isEmpty {
            throw AuthServiceError.usernameTaken
        }

        let response = try await supabase.auth.signUp(
            email: email,
            password: password,
            data: ["username": .string(username)]
        )

        // If session is nil, email confirmation is required
        return response.session != nil
    }

    // Sign out
    func signOut() async throws {
        try await supabase.auth.signOut()
    }
}

enum AuthServiceError: LocalizedError {
    case usernameTaken
    case invalidUsername

    var errorDescription: String? {
        switch self {
        case .usernameTaken: return "Username is already taken"
        case .invalidUsername: return "Username must be at least 3 characters (letters, numbers, underscores only)"
        }
    }
}
```
