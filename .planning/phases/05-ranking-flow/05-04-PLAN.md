```swift
private func saveBook() {
    guard !isSaving else { return }
    guard let category = category, let tier = tier, let position = finalPosition else { return }
    guard case .authenticated(let user) = authService.state else { return }

    isSaving = true

    Task {
        do {
            let params = RankBookParams(
                userId: user.id,
                openLibraryKey: bookKey,
                title: title,
                author: author,
                coverUrl: selectedCover ?? coverUrl,
                category: category,
                tier: tier,
                rankPosition: position,
                score: 0,  // Will be calculated by RPC
                reviewText: reviewText.isEmpty ? nil : reviewText,
                finishedAt: finishedAt,
                existingEntryId: existingEntry?.id
            )

            let score = try await RankingService.rankBook(params: params)

            // Fire-and-forget activity logging
            let metadata = BookMetadata(
                openLibraryKey: bookKey,
                title: title,
                author: author,
                coverUrl: selectedCover ?? coverUrl
            )
            Task { @Sendable in
                await RankingService.logRankingActivity(
                    userId: user.id,
                    book: metadata,
                    coverUrl: selectedCover ?? coverUrl,
                    score: score,
                    category: category
                )
            }

            // Success haptic
            UINotificationFeedbackGenerator().notificationOccurred(.success)

            // Dismiss
            dismiss()
        } catch {
            isSaving = false
            print("Error saving ranking: \(error)")
            // Could show alert, but for now just print
        }
    }
}
```

IMPORTANT: Read both files first before modifying.
IMPORTANT: RankBookParams may need adjustment — check the exact fields in UserBook.swift. The `score` field might not be needed in params (RPC calculates it). Adapt as needed.
IMPORTANT: The `existingEntryId` field in RankBookParams may be named differently — check UserBook.swift for the exact field name. In the web app it's `p_existing_entry_id`.
IMPORTANT: finishedAt is a Date in Swift but the RPC expects a date string. Format it or let Supabase handle the encoding.
IMPORTANT: After saving, dismiss() pops back to the book detail view since this is pushed via NavigationLink.
