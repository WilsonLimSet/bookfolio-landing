   CSV line parser (handle quotes):
   ```swift
   private static func parseCSVLine(_ line: String) -> [String] {
       var result: [String] = []
       var current = ""
       var inQuotes = false
       var chars = Array(line)
       var i = 0
       while i < chars.count {
           if inQuotes {
               if chars[i] == "\"" && i + 1 < chars.count && chars[i + 1] == "\"" {
                   current.append("\"")
                   i += 2
               } else if chars[i] == "\"" {
                   inQuotes = false
                   i += 1
               } else {
                   current.append(chars[i])
                   i += 1
               }
           } else {
               if chars[i] == "\"" {
                   inQuotes = true
                   i += 1
               } else if chars[i] == "," {
                   result.append(current.trimmingCharacters(in: .whitespaces))
                   current = ""
                   i += 1
               } else {
                   current.append(chars[i])
                   i += 1
               }
           }
       }
       result.append(current.trimmingCharacters(in: .whitespaces))
       return result
   }
   ```

b. `static func matchBooks(_ books: [GoodreadsBook], onProgress: @escaping (Int) -> Void) async -> [MatchedBook]`:

   - For each book: search OpenLibrary with "\(title) \(author)"
   - Take first result: openLibraryKey, coverUrl
   - Detect category via OpenLibraryService.detectCategory if key found
   - Fallback category from shelves using keyword matching (same as web app)
   - 500ms delay between batches (Task.sleep)
   - Call onProgress after each batch with count of matched so far

c. `static func calculateImportData(_ books: [MatchedBook]) -> [(book: MatchedBook, score: Double, tier: String, position: Int)]`:
   - Group by rating (5, 4, 3, 2, 1)
   - For each group, sort by title
   - Calculate tier: 4-5 = "liked", 3 = "fine", 1-2 = "disliked"
   - Score ranges: liked (6.7-10.0), fine (3.4-6.6), disliked (0.0-3.3)
   - Position within group: score = tierMin + (index / groupSize) * tierRange
   - Overall position: running count across all groups (highest rated first)
   - Return array with calculated data

d. `static func importBooks(userId: UUID, books: [(book: MatchedBook, score: Double, tier: String, position: Int)], onProgress: @escaping (Int) -> Void) async`:
